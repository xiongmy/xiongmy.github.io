如何用js读取字符串中每个字符重复的次数？

之前面试的时候曾经遇到过一个这样的问题，当时被卡住了，今天偶然在segmentfault看到了有人提了，细细的琢磨了下每个答案，收获颇丰，摘录在此，以备以后翻阅。原文链接

方法一

var arr = 'abcdaabc';
var info = arr
.split('')
.reduce((p, k) => (p[k] ? p[k]++ : p[k] = 1, p), {});
console.log(info);

方法二

var temp = {};
   'abcdaabc'.replace(/(\w{1})/g,function($1){
    temp[$1] ? temp[$1]+=1 : temp[$1] = 1;
});
console.log(temp);

方法三

const arr="abcdaabc";
let counter = {};
for (let i = 0, len = arr.length; i < len; i++ ) {
    counter[arr[i]] ? counter[arr[i]]++ : counter[arr[i]] = 1;
}
console.log(counter);

方法四

var str = "aasdkfjwoensweljoipwefslnvxcvasfs";

str = str.split('').sort().join('');

var arr = str.match(/(\w)\1+/g);

var result = {}

for(var i = 0, j = arr.length; i < j; i++){
    var item = arr[i];
    result[item.charAt(0)] = item.length;
}

console.table(result);
